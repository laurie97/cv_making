%----------------------------------------------------------------------------------------
%	JOB COMMAND
%----------------------------------------------------------------------------------------

\newcommand{\job}[6]{
\begin{tabbing}
\hspace{2cm} \= \kill
#1 \> \href{#4}{\textbf{#3}} \\
#2 \>\+ \textit{#5} \\
\hspace{-1.5cm}
\begin{minipage}{\smallertextwidth}
\vspace{2mm}
#6
\end{minipage}
\end{tabbing}
}

\newcommand{\edu}[4]{
\begin{tabbing}
\hspace{2cm} \= \kill
{#1} \>\+ \href{#3}{\textbf{#2}}\\
\hspace{-1.5cm}
\begin{minipage}{\smallertextwidth}
\vspace{1mm}
#4
\end{minipage}
\end{tabbing}
}
